  2. Working with Structures.

    • Write a definition for a struct ApptPage (appointment page). Each ApptPage should have a field for the date and a field for the number of appointments. (3 points)

      	struct ApptPage {
	   Date date;
	   int num_appts;
	};

    • Write a code segment that declares an appointment page variable, sets the date to October 31 and sets the number of appointments to 5. 

      	ApptPage ap;
	ap.date.month = OCT;
	ap.date.day = 31;
	ap.num_appts = 5;

      or

      	ApptPage ap = {{OCT, 31}, 5};

    • Write a function that takes a date and returns a logical value based on whether that date is valid or not. 

      	int validDate(Date d) {

	   int valid;  // 1 if valid, 0 if invalid

	   switch (d.month) {
	      case JAN:
	      case MAR:
	      case MAY:
	      case JUL:
	      case AUG:
	      case OCT:
	      case DEC:
	         valid = (d.day >= 1 && d.day <= 31); break;
	      case APR:
	      case JUN:
	      case SEP:
	      case NOV:
		 valid = (d.day >= 1 && d.day <= 30); break;
	      case FEB:
 	         valid = (d.day >= 1 && d.day <= 29); break;
	      default:
		 valid = 0;   // invalid Month!
           }

           return (valid);
	}
